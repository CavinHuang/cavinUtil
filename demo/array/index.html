array demo

      
        const array = require('array');

        // arrayEqual demo
        let arr1 = [1, 2, '11'];
        let arr2 = [1, 2, '11'];
        let arr3 = [1, 2, '11', 2];

        console.log(array.arrayEqual(arr1, arr2)); // true
        console.log(array.arrayEqual(arr2, arr3)); // false

        // concat
        console.log(array.concat(arr1, arr2, arr3)); // [1, 2, "11"]

        // inArray
        console.log(array.inArray(2, arr3)); // 1
        console.log(array.inArray(2, arr3, 3)); // 3

        // isArray
        console.log(array.isArray(arr1)); // true

        // join
        console.log(array.join('_', arr1, arr2, arr3)); // 1_2_11_1_2_11_1_2_11_2

        // unique
        console.log(array.unique(arr3)); // [1, "11", 2]
